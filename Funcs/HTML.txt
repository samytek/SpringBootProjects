<div>
    <label for="email">Enter Email:</label>{{emailId}}
    <input 
      id="email" 
      type="email" 
      [(ngModel)]="emailId" 
      (ngModelChange)="onEmailChange()"
      placeholder="Enter your email"
    />
    <div *ngIf="isInvalidEmail" style="color: red;">
      Invalid email. Please use a labcorp.com email address.
    </div>
  </div>
  
  emailId: string = '';
  isInvalidEmail: boolean = false;

  onEmailChange() {
    const emailRegex = /^[a-zA-Z0-9._%+-]+@labcorp\.com$/;
	const emailRegex = /^[a-zA-Z0-9.]+@labcorp\.com$/;
    this.isInvalidEmail = !emailRegex.test(this.emailId);
  }

  getStateSearchDetails(val) {
    var searchText = val.trim();
    this.masterDataList = $.map(this.originalStateDetails, function (val, key) {
      if (val.bankName.toLowerCase().includes(searchText.toLowerCase()) ||
        val.branchCode.toLowerCase().includes(searchText.toLowerCase()) ||
        val.branchName.toLowerCase().includes(searchText.toLowerCase()) ||
        val.ifsc.toLowerCase().includes(searchText.toLowerCase())
      ) {
        return val;
      }
    });
    this.search();
  }
